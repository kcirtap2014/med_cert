def subsampling(df, fraction = 0.2):
    
    train_sample = int(0.2*df[df.type == 'train'].shape[0])
    valid_sample = int(0.2*df[df.type == 'valid'].shape[0])
    test_sample = int(0.2*df[df.type == 'test'].shape[0])

    cols = []
    cols.append(df[df.type == 'train'].sample(train_sample))
    cols.append(df[df.type == 'valid'].sample(valid_sample))
    cols.append(df[df.type == 'test'].sample(test_sample))
    
    df_sub = pd.concat(cols, axis=0).reset_index().drop(columns=['index'])
    
    return df_sub

def train_test_split(df, X_columns, y_column):
    
    X_train = df[df.type == 'train'][X_columns]
    X_valid = df[df.type == 'valid'][X_columns]
    X_test = df[df.type == 'test'][X_columns]
    
    y_train = df.loc[X_train.index][y_column]
    y_valid = df.loc[X_valid.index][y_column]
    y_test = df.loc[X_test.index][y_column]
    
    return X_train, y_train, X_valid, y_valid, X_test, y_test

def image_preprocessing(img):
    mat_img = np.array(img)

    # increase contrast
    pxmin = np.min(mat_img)
    pxmax = np.max(mat_img)
    mat_img = ((mat_img - pxmin) / (pxmax - pxmin) * 255).astype(np.uint8)
    img2 = copy(mat_img)
    img2, img = thresholding(img, img2, DIR_PATH, option = 1)
    mat_img = np.asarray(img)
    
    return mat_img
